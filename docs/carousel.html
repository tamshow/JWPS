<script>
    var swiperTop = new Swiper('#js-swiper-top', {
      pagination: '.swiper-top .swiper-pagination',
      paginationClickable: true,
      centeredSlides: false,
      loop: true,
      loopAdditionalSlides: 1,
      spaceBetween: 0,
      slidesPerView: 1,
      autoplay: 5000,
      autoHeight: true,
      effect: 'fade',
      breakpoints: {
        599: {
          onInit() {
            $(window).on('load resize', function () {
              mainVisualHeight();
            });
            function mainVisualHeight() {
              var windowH = $(window).height(),
                  headerH = $('.l-header').height();
              $('#js-swiper-top .swiper-slide').each(function(index) {
                $('.swiper-slide__item1 img', this).height(windowH - headerH - $('.swiper-slide__item2', this).outerHeight());
              });
            }
          }
        }
      },
      onInit() {
        var start = $('.swiper-top .swiper-start');
        var stop = $('.swiper-top .swiper-stop');
        start.on('click', function() {
          if (!start.hasClass('on')) {
            swiperTop.startAutoplay();
            start.addClass('on');
            stop.removeClass('on');
          }
        });
        stop.on('click', function() {
          if (!stop.hasClass('on')) {
            swiperTop.stopAutoplay();
            stop.addClass('on');
            start.removeClass('on');
          }
        });
      }
    });



</script>
